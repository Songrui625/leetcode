class Solution {
    public int maxArea(int[] height) {
        int len = height.length;
        int res = 0;
        int area = 0;
        for (int i = 0; i <= len; i++) {
            for (int j = i + 1; j < len; j++) {
                area = (j-i) * Math.min(height[i], height[j]);
                if (res < area) {
                    res  = area; 
                }
            }
        }
        return res;
    }
}
